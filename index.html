const SHEET_ID = '1m3N7OL07UkKzCfrMeylr2jkPgkQMT4kRNRCqvMCEGuo'; function doGet(e) { const result = handleRequest(e); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const result = handleRequest(e); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function handleRequest(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const ss = SpreadsheetApp.openById(SHEET_ID); const action = e.parameter.action || (e.postData ? JSON.parse(e.postData.contents).action : ''); const data = e.postData ? JSON.parse(e.postData.contents) : {}; let result = {}; if (action === 'getAll') { result = { workers: sheetToJson(ss,'Trabajadores'), centros: sheetToJson(ss,'Centros'), asistencias: getAsistencias(ss), config: getConfig(ss) }; } else if (action === 'saveWorkers') { jsonToSheet(ss,'Trabajadores',data.workers); result = { ok: true }; } else if (action === 'saveCentros') { jsonToSheet(ss,'Centros',data.centros); result = { ok: true }; } else if (action === 'saveAsistencia') { saveAsistencia(ss,data.wid,data.fecha,data.rec); result = { ok: true }; } else if (action === 'saveConfig') { saveConfig(ss,data.config); result = { ok: true }; } return result; } finally { lock.releaseLock(); } } function sheetToJson(ss, name) { const sheet = ss.getSheetByName(name); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(function(row) { const obj = {}; headers.forEach(function(h,i) { obj[h] = row[i]; }); return obj; }); } function jsonToSheet(ss, name, arr) { const sheet = ss.getSheetByName(name); sheet.clearContents(); if (!arr || !arr.length) return; const headers = Object.keys(arr[0]); sheet.getRange(1,1,1,headers.length).setValues([headers]); const rows = arr.map(function(obj) { return headers.map(function(h) { return obj[h] || ''; }); }); sheet.getRange(2,1,rows.length,headers.length).setValues(rows); } function getAsistencias(ss) { const sheet = ss.getSheetByName('Asistencias'); if (!sheet) return {}; const data = sheet.getDataRange().getValues(); if (data.length < 2) return {}; const result = {}; data.slice(1).forEach(function(row) { const wid = row[0], fecha = row[1], tipo = row[2], centro = row[3]; if (!result[wid]) result[wid] = {}; result[wid][String(fecha)] = { tipo: tipo, centro: centro }; }); return result; } function saveAsistencia(ss, wid, fecha, rec) { const sheet = ss.getSheetByName('Asistencias'); const data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]) === String(wid) && String(data[i][1]) === String(fecha)) { sheet.getRange(i+1,3,1,2).setValues([[rec.tipo||'', rec.centro||'']]); return; } } sheet.appendRow([wid, fecha, rec.tipo||'', rec.centro||'']); } function getConfig(ss) { const sheet = ss.getSheetByName('Config'); const data = sheet.getDataRange().getValues(); const obj = {}; data.forEach(function(row) { if (row[0]) obj[row[0]] = row[1]; }); return obj; } function saveConfig(ss, config) { const sheet = ss.getSheetByName('Config'); sheet.clearContents(); Object.keys(config).forEach(function(k) { sheet.appendRow([k, config[k]]); }); }
